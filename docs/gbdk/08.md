# ゲームボーイで動くプログラムを書いてみる

## 08. 実践3の解答例

### 実践3の解答例
```c
#include <stdio.h>
#include <gb/gb.h>
#include <gb/drawing.h>

int main() {
    UINT8 joypad_result;

    while (1) {
       
        joypad_result = joypad();

        if (joypad_result == J_A){
            gotogxy(16,8);
            gprintf("A");
        } else {
            gotogxy(16,8);
            gprintf(" ");
        }

        if (joypad_result == J_B){
            gotogxy(14,10);
            gprintf("B");
        } else {
            gotogxy(14,10);
            gprintf(" ");
        }

        if (joypad_result == J_UP){
            gotogxy(3,7);
            gprintf("%c", 1);
        } else {
            gotogxy(3,7);
            gprintf(" ");
        }

        if (joypad_result == J_DOWN){
            gotogxy(3,11);
            gprintf("%c", 2);
        } else {
            gotogxy(3,11);
            gprintf(" ");
        }

        if (joypad_result == J_RIGHT){
            gotogxy(5,9);
            gprintf("%c", 3);
        } else {
            gotogxy(5,9);
            gprintf(" ");
        }
        
        if (joypad_result == J_LEFT){
            gotogxy(1,9);
            gprintf("%c" ,4);
        } else {
            gotogxy(1,9);
            gprintf(" ");
        }

        if (joypad_result == J_SELECT){
            gotogxy(4,15);
            gprintf("SELECT");
        } else {
            gotogxy(4,15);
            gprintf("      ");
        }

        if (joypad_result == J_START){
            gotogxy(11,15);
            gprintf("START");
        } else {
            gotogxy(11,15);
            gprintf("     ");
        }
    }
}
```
